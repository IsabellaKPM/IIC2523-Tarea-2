Nodo_UWU;Nodo_A;Nodo_B;Nodo_C;Nodo_D    # 5 nodos en total
1;2;3;4;5	    			            # D y C tiene ID más alto. Serán proponentes

# Empezamos los eventos

####################################
############ FASE 1 - Livelock hasta matar 1 proponente
####################################

Prepare;Nodo_C;1
Prepare;Nodo_D;2

Accept;Nodo_C;1;SET-ideas-0     # Rechazado porque existe ID=2
Prepare;Nodo_C;3                # Sube el ID a 3

Accept;Nodo_D;2;SET-ideas-1     # Rechazado porque existe ID=3
Prepare;Nodo_D;4                # Sube el ID a 4

Accept;Nodo_C;3;SET-ideas-2     # Rechazado porque existe ID=4
Prepare;Nodo_C;5                # Sube el ID a 5

Accept;Nodo_D;4;SET-ideas-3    # Rechazado porque existe ID=5
Prepare;Nodo_D;6                # Sube el ID a 6

Stop;Nodo_C;Nodo_A
# Matamos al Nodo C para que deje de molestar
# Nodo B pasa a ser proponentes. Queda 1 aceptante (de 3 iniciales)

Start;Nodo_C
# Vuelve nodo C pero ahora como aceptante. Hay 2 de 3 aceptantes.


Accept;Nodo_D;6;SET-se_me_acabaron_las_ideas-Sí
# Nodo UWU recibe sin problema
# Nodo C acepta esa operación porque entiende si se envió el accept es porque 
# el Prepare correspondiente logró mayoría antes.

Learn
# Se consolida SET-se_me_acabaron_las_ideas-Sí

####################################
############ FASE 1 - No se logra consenso y en otro si se logra con identificadores distintos.
####################################

Start;Nodo_A
# Ahora están todos activos. B y C son proponentes

Prepare;Nodo_B;14
# Recibe 3 de 3 ok

Stop;Nodo_A;Nodo_D
# Solo queda Nodo_UWU como aceptante.

Accept;Nodo_B;14;SET-se_me_acabaron_las_ideas-O tal vez no
# Solo Nodo_UWU recibe esta operación con ID=14

Stop;Nodo_UWU
Start;Nodo_D;Nodo_A
# 2 de 3 nodos aceptantes activos que no tienen ninguna operación

Prepare;Nodo_B;17
# 2 de 3 ok

Stop;Nodo_D
# Solo queda Nodo_A como aceptante.

Accept;Nodo_B;17;SET-se_me_acabaron_las_ideas-efectivamente se me acabaron
# Solo Nodo_A recibe esta operación con ID=17

Start;Nodo_D;Nodo_UWU
# Todos los aceptantes activos ahora

Prepare;Nodo_C;24
# Se entera de las 2 operaciones ya aceptadas por Nodo_A y Nodo_UWU
# Se queda con la mayor, la de ID=17 (SET-se_me_acabaron_las_ideas-efectivamente se me acabaron)

Stop;Nodo_A;Nodo_UWU
# Solo queda Nodo_D como aceptante

# Bifuración!!!
*Accept;Nodo_C;24;SET-se_me_acabaron_las_ideas-[PENDIENTE]
# Simulación 1: Nodo_D recibe 'SET-se_me_acabaron_las_ideas-efectivamente se me acabaron' con ID=24
# Simulación 2: Nodo_D no recibe nada.

Start;Nodo_A;Nodo_UWU
# En ambas simulaciones quedan todos prendidos.

Learn
# Simulación 1: 2 nodos (A y D) tienen la misma operación (aunque con distinto ID). 
#   --> Logra consenso y por lo tanto la variable "se_me_acabaron_las_ideas" se sobreescribe.

# Simulación 2: Nodo D no tiene nada, A y UWU tienen operaciones distintas (el valor es distinto). 
#   --> No logra consenso y por lo tanto la variable "se_me_acabaron_las_ideas" no se sobreescribe
